import { useState } from 'react'

function ShoppingCart() {
  const [items, setItems] = useState([
    { id: 1, name: 'Laptop', price: 50000, quantity: 1 },
    { id: 2, name: 'Phone', price: 30000, quantity: 2 }
  ])

  const handleAddItem = () => {
    const newItem = {
      id: Date.now(),
      name: 'New Item',
      price: 10000,
      quantity: 1
    }
    setItems([...items, newItem])
  }

  const handleRemoveItem = (id) => {
    setItems(items.filter(item => item.id !== id))
  }

  const handleIncreaseQuantity = (id) => {
    setItems(items.map(item =>
      item.id === id ? { ...item, quantity: item.quantity + 1 } : item
    ))
  }

  const handleDecreaseQuantity = (id) => {
    setItems(items.map(item =>
      item.id === id && item.quantity > 1 
        ? { ...item, quantity: item.quantity - 1 } 
        : item
    ))
  }

  const totalPrice = items.reduce((sum, item) => sum + (item.price * item.quantity), 0)

  return (
    <div style={{ maxWidth: '600px', margin: '20px auto', padding: '20px' }}>
      <h2>🛒 Shopping Cart</h2>

      {items.map(item => (
        <div
          key={item.id}
          style={{
            display: 'flex',
            justifyContent: 'space-between',
            alignItems: 'center',
            padding: '12px',
            border: '1px solid #ddd',
            marginBottom: '10px',
            borderRadius: '5px'
          }}
        >
          <div style={{ flex: 1 }}>
            <strong>{item.name}</strong>
            <p style={{ margin: '4px 0 0 0', color: '#999' }}>₹{item.price}</p>
          </div>

          <div style={{ display: 'flex', alignItems: 'center', gap: '10px', marginRight: '10px' }}>
            <button onClick={() => handleDecreaseQuantity(item.id)} style={{ padding: '5px 10px' }}>-</button>
            <span style={{ fontWeight: 'bold', minWidth: '30px', textAlign: 'center' }}>{item.quantity}</span>
            <button onClick={() => handleIncreaseQuantity(item.id)} style={{ padding: '5px 10px' }}>+</button>
          </div>

          <strong style={{ marginRight: '10px', minWidth: '80px', textAlign: 'right' }}>
            ₹{item.price * item.quantity}
          </strong>

          <button
            onClick={() => handleRemoveItem(item.id)}
            style={{ padding: '5px 10px', background: 'red', color: 'white', border: 'none', borderRadius: '3px', cursor: 'pointer' }}
          >
            🗑️
          </button>
        </div>
      ))}

      <button
        onClick={handleAddItem}
        style={{
          width: '100%',
          padding: '10px',
          background: 'green',
          color: 'white',
          border: 'none',
          borderRadius: '5px',
          marginTop: '10px',
          cursor: 'pointer'
        }}
      >
        ➕ Add Item
      </button>

      <div style={{ marginTop: '20px', padding: '15px', background: '#f0f0f0', borderRadius: '5px' }}>
        <h3 style={{ marginTop: 0 }}>💰 Total: ₹{totalPrice}</h3>
      </div>
    </div>
  )
}

export default ShoppingCart
