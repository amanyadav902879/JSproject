function ProductList() {
  const products = [
    { id: 1, name: 'Laptop', price: 50000, stock: true },
    { id: 2, name: 'Phone', price: 30000, stock: true },
    { id: 3, name: 'Tablet', price: 20000, stock: false },
    { id: 4, name: 'Headphones', price: 5000, stock: true }
  ]

  return (
    <div style={{ padding: '20px' }}>
      <h2>🛍️ Products</h2>
      <div style={{ display: 'grid', gridTemplateColumns: 'repeat(auto-fit, minmax(200px, 1fr))', gap: '15px' }}>
        {products.map(product => (
          <div
            key={product.id}
            style={{
              border: '1px solid #ddd',
              padding: '15px',
              borderRadius: '8px',
              background: product.stock ? 'white' : '#f5f5f5',
              opacity: product.stock ? 1 : 0.7
            }}
          >
            <h3>{product.name}</h3>
            <p>₹{product.price}</p>
            <p style={{ color: product.stock ? 'green' : 'red' }}>
              {product.stock ? '✅ In Stock' : '❌ Out of Stock'}
            </p>
            <button
              disabled={!product.stock}
              style={{
                width: '100%',
                padding: '8px',
                background: product.stock ? 'blue' : 'gray',
                color: 'white',
                border: 'none',
                borderRadius: '5px',
                cursor: product.stock ? 'pointer' : 'not-allowed'
              }}
            >
              {product.stock ? 'Add to Cart' : 'Unavailable'}
            </button>
          </div>
        ))}
      </div>
    </div>
  )
}

export default ProductList
